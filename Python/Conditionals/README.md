# Python Conditionals (Syntax Only)

## if – else

if condition:
    pass
else:
    pass

## if – elif – else

if condition1:
    pass
elif condition2:
    pass
elif condition3:
    pass
else:
    pass

## Nested if

if condition1:
    if condition2:
        pass
    else:
        pass
else:
    pass

## Nested if with elif

if condition1:
    if condition2:
        pass
    elif condition3:
        pass
    else:
        pass
else:
    if condition4:
        pass
    else:
        pass
